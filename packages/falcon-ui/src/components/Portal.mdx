import { Playground } from 'docz';
import { Toggle } from 'react-powerplug';
import { Portal, Sidebar, H1, H2, Backdrop, Button } from '@deity/falcon-ui';

<H2 my="lg" children="Portal" />

`Portal` component is a wrapper around [React Portals](https://reactjs.org/docs/portals.html)
and can be used in combination with other `falcon-ui` components (like `Sidebar` or `Backdrop` for example).
Components rendered  as `Portal` render into a new DOM node (`<div>`) that exists outside the DOM hierarchy of the parent component
and gets appended to `<body>`.

<Playground>
  <Toggle initial={false}>
    {({ on, toggle }) => (
      <React.Fragment>
        <Button onClick={toggle}>Show sidebar that renders as portal</Button>
        <Sidebar side="right" as={Portal} visible={on}>
          <H1 p="lg">Hello!</H1>
        </Sidebar>
        <Backdrop
          onClick={toggle}
          as={Portal}
          visible={on}
        />
      </React.Fragment>
    )}
  </Toggle>
</Playground>
